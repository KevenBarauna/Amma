CREATE TABLE USUARIO
(
    [id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [nome] VARCHAR(30) NOT NULL, 
    [senha] VARCHAR(15)  NULL, 
    [tema] VARCHAR(1) NOT NULL,
    [id_avatar] VARCHAR(3) NOT NULL, 
    [cargo] VARCHAR(20) NOT NULL, 
)

CREATE TABLE VOTOS_USUARIO
(
    [id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [id_ticket] INT NOT NULL, 
    [id_usuario] INT NOT NULL, 
    [data] VARCHAR(10) NOT NULL, 

)

CREATE TABLE TIPO_TICKET
(
    [id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [titulo] VARCHAR(20) NOT NULL, 
)

CREATE TABLE TICKET
(
    [id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [id_tipo] INT NOT NULL, 
    [id_usuario] INT NOT NULL, 
    [titulo] VARCHAR(30) NOT NULL, 
    [descricao] VARCHAR(100) NULL, 
    [data] VARCHAR(10) NOT NULL, 
    [identificar] VARCHAR(1) NOT NULL,
    [qtd_votos] INT NOT NULL, 
    [ind_concluido] VARCHAR(1) NOT NULL,
    [ind_apagado] VARCHAR(1) NOT NULL,
)

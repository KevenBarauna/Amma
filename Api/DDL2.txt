CREATE TABLE usuario
(
[id] integer NOT NULL,
[nome] string(40) NOT NULL,
[senha] string(30) NOT NULL,
[email] string(35),
[cargo] string(35),
[idAvatar] integer NOT NULL,
[idPermissao] integer NOT NULL,
)

CREATE TABLE sugestoes
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[idUsuario] INT NOT NULL, 
[idCategoria] INT NOT NULL, 
[idStatus] INT NOT NULL, 
[anonimo] VARCHAR(1)DEFAULT 'N',
[titulo] VARCHAR(30) NOT NULL, 
[descricao] VARCHAR(90) NOT NULL, 
[quantidadeVotos] INT NOT NULL, 
[data] DATE,
[notificacao] VARCHAR(1)DEFAULT 'N',
)

CREATE TABLE status
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[descricao] VARCHAR(20) NOT NULL, 
)

CREATE TABLE categoria
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[descricao] VARCHAR(10) NOT NULL, 
)

CREATE TABLE permissao
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[descricao] VARCHAR(10) NOT NULL, 
)

CREATE TABLE usuario_votos
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[idUsuario] INT NOT NULL, 
[idSugestao] INT NOT NULL, 
)
